    }, (err) => console.error("Conf listener error:", err));

      unsubConfs();

      status: 'Submitted',

      note: 'Resubmission'

    };

    const docRef = doc(db, 'artifacts', appId, 'public', 'data', 'journals', journalId);

      status: newStatus

                        }`}

                  </div>

                ))

            </div>

        )}

      {showJournalModal && (

              </div>

              <div>

                <label className="block text-xs font-bold text-slate-500 uppercase mb-2">所有作者 Email</label>

                <textarea required name="emails" rows="2" className="w-full px-4 py-3 bg-slate-50 border-slate-200 rounded-xl outline-none focus:ring-2 focus:ring-blue-500" />

              </div>

              <div className="grid grid-cols-2 gap-4">

                <div>

                  <label className="block text-xs font-bold text-slate-500 uppercase mb-2">初次投稿期刊</label>

                  <input required name="journal" className="w-full px-4 py-3 bg-slate-50 border-slate-200 rounded-xl outline-none focus:ring-2 focus:ring-blue-500" />

                </div>

                <div>

                  <input required name="date" type="date" defaultValue={new Date().toISOString().split('T')[0]} className="w-full px-4 py-3 bg-slate-50 border-slate-200 rounded-xl outline-none focus:ring-2 focus:ring-blue-500" />

                </div>

              </div>

              <div className="flex gap-3 pt-6">

                <button type="button" onClick={() => setShowJournalModal(false)} className="flex-1 py-3 text-slate-600 font-bold hover:bg-slate-100 rounded-xl">取消</button>

                <button type="submit" className="flex-1 py-3 bg-blue-600 text-white rounded-xl font-bold">建立紀錄</button>

              </div>

            </form>

          </div>

        </div>

      )}


      {/* 研討會彈窗 */}

      {showConfModal && (

              </div>

              <div className="grid grid-cols-2 gap-4">

                  <label className="block text-xs font-bold text-slate-500 uppercase mb-2">地點</label>

                  <input required name="location" className="w-full px-4 py-3 bg-slate-50 border-slate-200 rounded-xl outline-none focus:ring-2 focus:ring-blue-500" />

                </div>

                  <label className="block text-xs font-bold text-slate-500 uppercase mb-2">舉辦日期</label>

                </div>

              </div>

              <div className="grid grid-cols-2 gap-4">

              </div>

              <div className="flex gap-3 pt-6">

                <button type="button" onClick={() => setShowConfModal(false)} className="flex-1 py-3 text-slate-600 font-bold hover:bg-slate-100 rounded-xl">取消</button>

            </form>

          </div>

        </div>

      )}

  );
